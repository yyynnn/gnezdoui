## Controlled & Clearable

<Playground>
  {() => {
    const [value, setValue] = useState('')
    return (
      <React.Fragment>
        <MoneyInput
          placeholder="Введите сумму"
          clear={true}
          onClear={() => {
            setValue('')
          }}
          value={value}
          onChange={e => {
            setValue(e.currentTarget.value)
          }}
        />
        <Text>Input value: {value}</Text>
      </React.Fragment>
    )
  }}
</Playground>
