    const [value, setValue] = useState(0)
    return (
      <React.Fragment>
        <NumberInput
          value={value}
          onChange={value => {
            setValue(value)
          }}
        />
        <Text>Input value: {value}</Text>
      </React.Fragment>
    )
  }}
