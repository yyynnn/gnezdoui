# Input Slider&Range

## Basic

<Playground>
  {() => {
    const [value, setValue] = useState(50)
    return (
      <SliderInput
        label={'Сумма кредита'}
        tooltip={'Привет, мир!'}
        min={1}
        max={100}
        value={value}
        onChange={setValue}
      />
    )
  }}
</Playground>
