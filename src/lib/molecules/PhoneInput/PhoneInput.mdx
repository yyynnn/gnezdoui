## Controlled & Clearable

<Playground>
  {() => {
    const [value, setValue] = useState('+7 ')
    return (
      <React.Fragment>
        <PhoneInput
          label="Номер телефона: "
          clear={true}
          onClear={() => {
            setValue('+7')
          }}
          value={value}
          onChange={e => {
            setValue(e.currentTarget.value)
          }}
        />
        <Text>Input value: {value}</Text>
      </React.Fragment>
    )
  }}
</Playground>
