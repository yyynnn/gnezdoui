## Controlled & Clearable

<Playground>
  {() => {
    const [value, setValue] = useState('')
    return (
      <React.Fragment>
        <Input
          label="Фамилия, имя и отчество"
          placeholder={'Зубенко Михаил Петрович'}
          clear={true}
          onClear={() => {
            setValue('')
          }}
          value={value}
          onChange={e => {
            setValue(e.currentTarget.value)
          }}
        />
        <Text>Input value: {value}</Text>
      </React.Fragment>
    )
  }}
</Playground>
